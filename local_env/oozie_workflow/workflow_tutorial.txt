<?xml version="1.0" encoding="UTF-8"?>
<workflow-app name="Stock_BigData_Pipeline_Advanced" xmlns="uri:oozie:workflow:0.4">

    <start to="sqoop-import-node"/>

    <action name="sqoop-import-node">
        <sqoop xmlns="uri:oozie:sqoop-action:0.4">
            <job-tracker>${jobTracker}</job-tracker>
            <name-node>${nameNode}</name-node>
            <prepare>
                <delete path="${nameNode}/user/quangduy/raw_data"/>
            </prepare>
            
            <arg>import</arg>
            <arg>--connect</arg>
            <arg>jdbc:mysql://localhost:3306/bigdata_stock</arg>
            <arg>--username</arg>
            <arg>root</arg>
            <arg>--password</arg>
            <arg>password123</arg>
            <arg>--table</arg>
            <arg>tbl_raw_stock</arg>
            <arg>--target-dir</arg>
            <arg>/user/quangduy/raw_data</arg>
            <arg>-m</arg>
            <arg>1</arg>
        </sqoop>
        <ok to="hive-clean-node"/>
        <error to="fail"/>
    </action>

    <action name="hive-clean-node">
        <hive xmlns="uri:oozie:hive-action:0.2">
            <job-tracker>${jobTracker}</job-tracker>
            <name-node>${nameNode}</name-node>
            <prepare>
                <delete path="${nameNode}/user/quangduy/cleaned_data"/>
            </prepare>
            <job-xml>/user/hadoop/scripts/hive-site.xml</job-xml>
            <script>/user/hadoop/scripts/clean_data_advanced.hql</script>
            <param>INPUT=/user/quangduy/raw_data</param>
            <param>OUTPUT=/user/quangduy/cleaned_data</param>
        </hive>
        <ok to="mapreduce-python-node"/>
        <error to="fail"/>
    </action>

    <action name="mapreduce-python-node">
        <map-reduce>
            <job-tracker>${jobTracker}</job-tracker>
            <name-node>${nameNode}</name-node>
            <prepare>
                <delete path="${nameNode}/user/hadoop/stock_result"/>
            </prepare>
            <configuration>
                <property>
                    <name>mapred.mapper.class</name>
                    <value>org.apache.hadoop.mapred.lib.IdentityMapper</value>
                </property>
                <property>
                    <name>mapred.reducer.class</name>
                    <value>org.apache.hadoop.mapred.lib.IdentityReducer</value>
                </property>
                <property>
                    <name>mapred.input.dir</name>
                    <value>/user/quangduy/cleaned_data</value>
                </property>
                <property>
                    <name>mapred.output.dir</name>
                    <value>/user/hadoop/stock_result</value>
                </property>
                <property>
                    <name>stream.map.streamprocessor</name>
                    <value>python3 mapper.py</value>
                </property>
                <property>
                    <name>stream.reduce.streamprocessor</name>
                    <value>python3 reducer.py</value>
                </property>
            </configuration>
            <file>/user/hadoop/scripts/mapper.py#mapper.py</file>
            <file>/user/hadoop/scripts/reducer.py#reducer.py</file>
        </map-reduce>
        <ok to="kafka-stream-node"/>
        <error to="fail"/>
    </action>

    <action name="kafka-stream-node">
        <shell xmlns="uri:oozie:shell-action:0.2">
            <job-tracker>${jobTracker}</job-tracker>
            <name-node>${nameNode}</name-node>
            <exec>python3</exec>
            <argument>kafka_producer.py</argument>
            <argument>/user/hadoop/stock_result</argument>
            <file>/user/hadoop/scripts/kafka_producer.py#kafka_producer.py</file>
        </shell>
        <ok to="sqoop-backup-node"/>
        <error to="fail"/>
    </action>

    <action name="sqoop-backup-node">
        <sqoop xmlns="uri:oozie:sqoop-action:0.4">
            <job-tracker>${jobTracker}</job-tracker>
            <name-node>${nameNode}</name-node>
            <prepare>
                <delete path="${nameNode}/user/hadoop/stock_backup"/>
            </prepare>
            
            <arg>import</arg>
            <arg>--connect</arg>
            <arg>jdbc:mysql://localhost:3306/bigdata_stock</arg>
            <arg>--username</arg>
            <arg>root</arg>
            <arg>--password</arg>
            <arg>password123</arg>
            <arg>--table</arg>
            <arg>tbl_stock_analysis</arg>
            <arg>--target-dir</arg>
            <arg>/user/hadoop/stock_backup</arg>
            <arg>-m</arg>
            <arg>1</arg>
        </sqoop>
        <ok to="end"/>
        <error to="fail"/>
    </action>

    <kill name="fail">
        <message>Lỗi! Hệ thống Big Data dừng ở Node: [${wf:errorMessage(wf:lastErrorNode())}]</message>
    </kill>

    <end name="end"/>

</workflow-app>
